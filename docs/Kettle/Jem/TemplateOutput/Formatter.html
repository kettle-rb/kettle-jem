def phase(emoji, message, detail: nil)
  line = detail ? "[kettle-jem] #{emoji}  #{message} - #{detail}" : "[kettle-jem] #{emoji}  #{message}"
  @cli_io.puts(line)
  report_line(line)
  @phase_results << {emoji: emoji, message: message, detail: detail}
end
